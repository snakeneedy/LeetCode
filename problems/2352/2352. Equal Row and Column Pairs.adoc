class Solution:
    def equalPairs(self, grid: List[List[int]]) -> int:
        rowFreq = defaultdict(int)
        for row in grid:
            rowFreq[tuple(row)] += 1
        result = 0
        n = len(grid)
        for i in range(n):
            col = tuple(grid[j][i] for j in range(n))
            result += rowFreq[col]
        return result
